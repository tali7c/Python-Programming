\section{Core Concepts}
\label{sec:core}

\begin{frame}[fragile]{Learning Outcomes}
  \begin{itemize}[<+->]
    \item Explain what Pandas is used for
    \item Create and manipulate Series and DataFrames
    \item Use \texttt{head()}, \texttt{tail()}, column selection, and basic iteration
    \item Perform vectorized operations on columns
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Why Pandas?}
  \begin{itemize}[<+->]
    \item Works with tabular data (like Excel/CSV)
    \item Powerful cleaning and transformation functions
    \item Easy grouping/aggregation and missing value handling
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Series and DataFrame}
  \begin{itemize}[<+->]
    \item \textbf{Series}: 1D labeled array
    \item \textbf{DataFrame}: 2D table with labeled columns
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Create a DataFrame}
  \begin{lstlisting}[language=Python]
import pandas as pd

df = pd.DataFrame({
    "name": ["Asha", "Bilal", "Charu"],
    "marks": [88, 76, 91]
})
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Useful Methods}
  \begin{itemize}[<+->]
    \item \texttt{df.head(n)}, \texttt{df.tail(n)}
    \item \texttt{df.info()}, \texttt{df.describe()}
    \item Select a column: \texttt{df["marks"]}
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Add/Delete Columns}
  \begin{lstlisting}[language=Python]
df["cgpa"] = df["marks"] / 10
del df["cgpa"]
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Vectorized Operations}
  \begin{itemize}[<+->]
    \item Column math applies element-wise (fast)
  \end{itemize}
  \vspace{0.4em}
  \begin{lstlisting}[language=Python]
df["marks_plus_5"] = df["marks"] + 5
df["passed"] = df["marks"] >= 50
  \end{lstlisting}
\end{frame}
