\section{Core Concepts}
\label{sec:core}

\begin{frame}[fragile]{Learning Outcomes}
  \begin{itemize}[<+->]
    \item Create basic plots using Matplotlib pyplot
    \item Add titles, labels, legends, and grids
    \item Customize markers, line style, and colors
    \item Save plots to image files
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Matplotlib and \texttt{pyplot}}
  \begin{itemize}[<+->]
    \item Matplotlib is a plotting library for Python
    \item \texttt{matplotlib.pyplot} provides a simple plotting API
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Basic Line Plot}
  \begin{lstlisting}[language=Python]
import matplotlib.pyplot as plt

x = [1, 2, 3, 4]
y = [2, 4, 6, 8]

plt.plot(x, y)
plt.show()
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Labels, Title, Grid, Legend}
  \begin{lstlisting}[language=Python]
plt.plot(x, y, label="y = 2x")
plt.title("Line Plot")
plt.xlabel("x")
plt.ylabel("y")
plt.grid(True)
plt.legend()
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Saving a Plot}
  \begin{lstlisting}[language=Python]
plt.savefig("images/line_plot.png", dpi=150)
  \end{lstlisting}
\end{frame}
