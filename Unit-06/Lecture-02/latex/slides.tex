\section{Core Concepts}
\label{sec:core}

\begin{frame}[fragile]{Learning Outcomes}
  \begin{itemize}[<+->]
    \item Use important NumPy functions: \texttt{reshape}, \texttt{sum}, \texttt{mean}, \texttt{std}
    \item Generate arrays using \texttt{random}, \texttt{zeros}, \texttt{ones}, \texttt{empty}
    \item Sort and search arrays
    \item Compute dot products and matrix multiplication
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Reshaping Arrays}
  \begin{lstlisting}[language=Python]
a = np.arange(1, 10)
m = a.reshape((3, 3))
  \end{lstlisting}
  \begin{itemize}[<+->]
    \item Total elements must match (size stays same)
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Statistics}
  \begin{itemize}[<+->]
    \item \texttt{sum}, \texttt{mean}, \texttt{std}, \texttt{min}, \texttt{max}
    \item Use \texttt{axis=} for row/column-wise operations
  \end{itemize}
  \vspace{0.4em}
  \begin{lstlisting}[language=Python]
print(m.sum(axis=0))  # column sums
print(m.sum(axis=1))  # row sums
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Sorting and Searching}
  \begin{itemize}[<+->]
    \item \texttt{np.sort}, \texttt{np.argsort}
    \item \texttt{np.max}, \texttt{np.argmax}
    \item \texttt{np.where(condition)}
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Dot Product / Matrix Multiplication}
  \begin{lstlisting}[language=Python]
a = np.array([1, 2, 3])
b = np.array([4, 5, 6])
print(a.dot(b))

M = np.arange(1, 10).reshape((3, 3))
print(M @ M)
  \end{lstlisting}
\end{frame}
