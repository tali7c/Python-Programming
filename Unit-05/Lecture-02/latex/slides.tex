\section{Core Concepts}
\label{sec:core}

\begin{frame}[fragile]{Learning Outcomes}
  \begin{itemize}[<+->]
    \item Use public/private naming conventions in Python classes
    \item Inspect built-in class attributes like \texttt{\_\_dict\_\_} and \texttt{\_\_doc\_\_}
    \item Explain garbage collection at a high level
    \item Create abstract classes using the \texttt{abc} module
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Public vs Private in Python}
  \begin{itemize}[<+->]
    \item Python uses conventions (not strict access control)
    \item \texttt{name}: public
    \item \texttt{\_name}: internal (``protected'' by convention)
    \item \texttt{\_\_name}: name-mangled (harder to access accidentally)
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Built-in Class Attributes}
  \begin{itemize}[<+->]
    \item \texttt{\_\_dict\_\_}: attributes dictionary
    \item \texttt{\_\_doc\_\_}: documentation string
    \item \texttt{\_\_class\_\_}: object's class
    \item \texttt{\_\_module\_\_}: module name where class is defined
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Garbage Collection (GC)}
  \begin{itemize}[<+->]
    \item Automatically frees memory of unreachable objects
    \item Python uses reference counting + cycle detection
    \item Usually you do not manually free memory
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Abstract Classes}
  \begin{itemize}[<+->]
    \item Abstract class defines an interface (methods to implement)
    \item Prevents creating incomplete objects
  \end{itemize}
  \vspace{0.4em}
  \begin{lstlisting}[language=Python]
from abc import ABC, abstractmethod

class Shape(ABC):
    @abstractmethod
    def area(self):
        pass
  \end{lstlisting}
\end{frame}

\section{Summary}
\label{sec:summary}

\begin{frame}[fragile]{Key Takeaways}
  \begin{itemize}[<+->]
    \item ``Private'' in Python is mostly convention (\texttt{\_} and \texttt{\_\_})
    \item Built-in attributes help introspection (\texttt{\_\_dict\_\_}, \texttt{\_\_doc\_\_})
    \item GC manages memory automatically
    \item Abstract classes define required methods for subclasses
  \end{itemize}
\end{frame}
