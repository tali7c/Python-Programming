\section{Core Concepts}
\label{sec:core}

\begin{frame}[fragile]{Learning Outcomes}
  \begin{itemize}[<+->]
    \item Explain inheritance and its purpose
    \item Create subclasses and reuse parent methods/attributes
    \item Use \texttt{super()} to call parent constructor/methods
    \item Identify types of inheritance (single, multilevel, multiple, hierarchical)
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{What is Inheritance?}
  \begin{itemize}[<+->]
    \item Create a new class from an existing class
    \item Reuse common code and extend behavior
    \item ``is-a'' relationship (Student is a Person)
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Basic Syntax}
  \begin{lstlisting}[language=Python]
class Person:
    def __init__(self, name):
        self.name = name

class Student(Person):
    def __init__(self, name, sapid):
        super().__init__(name)
        self.sapid = sapid
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Types of Inheritance}
  \begin{itemize}[<+->]
    \item Single: A $\rightarrow$ B
    \item Multilevel: A $\rightarrow$ B $\rightarrow$ C
    \item Hierarchical: A $\rightarrow$ (B, C, D)
    \item Multiple: (A, B) $\rightarrow$ C
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Why \texttt{super()}?}
  \begin{itemize}[<+->]
    \item Calls parent class methods safely
    \item Avoids duplicating parent initialization code
    \item Important in multiple inheritance (method resolution order)
  \end{itemize}
\end{frame}

\section{Summary}
\label{sec:summary}

\begin{frame}[fragile]{Key Takeaways}
  \begin{itemize}[<+->]
    \item Inheritance reuses and extends code
    \item Subclasses can add new attributes and methods
    \item \texttt{super()} helps call parent logic correctly
    \item Multiple inheritance exists, but use it carefully
  \end{itemize}
