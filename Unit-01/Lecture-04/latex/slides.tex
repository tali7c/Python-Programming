\section{Core Concepts}
\label{sec:core}

\begin{frame}[fragile]{Learning Outcomes}
  \begin{itemize}[<+->]
    \item Read input and convert to correct types
    \item Format output using \texttt{print()} options and escape sequences
    \item Apply operators: arithmetic, relational, logical, bitwise, membership, identity
    \item Evaluate expressions using precedence and parentheses
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Input and Output}
  \begin{itemize}[<+->]
    \item \texttt{input(prompt)} reads text from keyboard (returns \texttt{str})
    \item \texttt{print(...)} displays output
    \item Useful \texttt{print} options: \texttt{sep}, \texttt{end}
  \end{itemize}
  \vspace{0.4em}
  \begin{lstlisting}[language=Python]
a = int(input("Enter a: "))
b = int(input("Enter b: "))
print("a =", a, "b =", b, sep=" | ")
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Escape Sequences}
  \begin{itemize}[<+->]
    \item New line: \texttt{\textbackslash n}
    \item Tab: \texttt{\textbackslash t}
    \item Backslash: \texttt{\textbackslash\textbackslash}
    \item Quotes inside strings: \texttt{\textbackslash"} or \texttt{\textbackslash'}
  \end{itemize}
  \vspace{0.4em}
  \begin{lstlisting}[language=Python]
print("Line1\nLine2")
print("A\tB\tC")
print("C:\\\\Users\\\\tofik.ali")
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Operator Categories}
  \small
  \begin{tabular}{ll}
    \toprule
    Category & Examples \\
    \midrule
    Arithmetic & \texttt{+ - * / \% // **} \\
    Relational & \texttt{< <= > >= == !=} \\
    Logical & \texttt{and or not} \\
    Bitwise & \texttt{\& | \string^ \string~ << >>} \\
    Membership & \texttt{in}, \texttt{not in} \\
    Identity & \texttt{is}, \texttt{is not} \\
    \bottomrule
  \end{tabular}
  \normalsize
\end{frame}

\begin{frame}[fragile]{Precedence (Why Parentheses Matter)}
  \begin{itemize}[<+->]
    \item Multiplication happens before addition
    \item Use parentheses to make intention clear
  \end{itemize}
  \vspace{0.4em}
  \begin{lstlisting}[language=Python]
print(2 + 3 * 4)     # 14
print((2 + 3) * 4)   # 20
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Membership and Identity}
  \begin{itemize}[<+->]
    \item Membership: check presence in a sequence/collection
    \item Identity: check if two names refer to the same object
  \end{itemize}
  \vspace{0.4em}
  \begin{lstlisting}[language=Python]
nums = [10, 20, 30]
print(20 in nums)  # True

a = [1, 2]
b = a
c = [1, 2]
print(a is b)      # True (same object)
print(a is c)      # False (different objects)
  \end{lstlisting}
\end{frame}
