\section{Core Concepts}
\label{sec:core}

\begin{frame}[fragile]{Learning Outcomes}
  \begin{itemize}[<+->]
    \item Use \texttt{pack} and \texttt{grid} layout managers correctly
    \item Organize widgets using Frames
    \item Create menu bars and menu commands
    \item Show message boxes and open file dialogs
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Layout Managers: \texttt{pack} vs \texttt{grid}}
  \begin{itemize}[<+->]
    \item \texttt{pack}: simple vertical/horizontal stacking
    \item \texttt{grid}: table-like rows and columns (forms)
    \item Rule: do not mix \texttt{pack} and \texttt{grid} in the same container
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Example: Simple \texttt{grid} Form}
  \begin{lstlisting}[language=Python]
tk.Label(root, text="Name").grid(row=0, column=0)
tk.Entry(root).grid(row=0, column=1)

tk.Label(root, text="Age").grid(row=1, column=0)
tk.Entry(root).grid(row=1, column=1)
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Frames for Better Structure}
  \begin{itemize}[<+->]
    \item Frame = container widget
    \item Helps grouping widgets (header, form, buttons, etc.)
    \item Makes layout easier and code more readable
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Menus}
  \begin{itemize}[<+->]
    \item A menu bar contains drop-down menus (File/Edit/Help)
    \item Each menu contains commands (Open, Save, Exit)
  \end{itemize}
  \vspace{0.4em}
  \begin{lstlisting}[language=Python]
menubar = tk.Menu(root)
file_menu = tk.Menu(menubar, tearoff=0)
file_menu.add_command(label="Exit", command=root.destroy)
menubar.add_cascade(label="File", menu=file_menu)
root.config(menu=menubar)
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Dialogs and Message Boxes}
  \begin{itemize}[<+->]
    \item Message boxes: info/warning/error/askyesno
    \item File dialogs: open/save file selection
  \end{itemize}
  \vspace{0.4em}
  \begin{lstlisting}[language=Python]
from tkinter import messagebox, filedialog
messagebox.showinfo("Done", "Saved successfully!")
path = filedialog.askopenfilename()
  \end{lstlisting}
\end{frame}

\section{Summary}
\label{sec:summary}

\begin{frame}[fragile]{Key Takeaways}
  \begin{itemize}[<+->]
    \item \texttt{pack} is good for simple stacking; \texttt{grid} is good for forms
    \item Frames improve structure and maintainability
    \item Menus provide standard UI navigation (File/Help)
    \item Message boxes and dialogs provide user feedback and file selection
  \end{itemize}
\end{frame}
