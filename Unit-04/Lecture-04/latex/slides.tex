\section{Core Concepts}
\label{sec:core}

\begin{frame}[fragile]{Learning Outcomes}
  \begin{itemize}[<+->]
    \item Explain why apps use databases instead of plain files
    \item Compare relational (SQL) and NoSQL databases
    \item Describe a basic GUI-to-database workflow
    \item Build a small Tkinter app that stores records in SQLite
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Why Databases?}
  \begin{itemize}[<+->]
    \item Data persistence across runs
    \item Fast search and structured storage
    \item Constraints (unique IDs), integrity, transactions
    \item Multi-user / multi-program access (in bigger systems)
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Relational vs NoSQL (Quick Comparison)}
  \small
  \begin{tabular}{ll}
    \toprule
    Relational (SQL) & NoSQL \\
    \midrule
    Tables (rows/columns) & Documents / key-value / graphs \\
    Fixed schema & Flexible schema \\
    SQL queries & Query APIs (varies) \\
    Example: SQLite, MySQL & Example: MongoDB \\
    \bottomrule
  \end{tabular}
  \normalsize
\end{frame}

\begin{frame}[fragile]{SQLite for Learning}
  \begin{itemize}[<+->]
    \item SQLite is embedded (no server)
    \item Uses one file as the database
    \item Great for demos and small apps
    \item Python has built-in \texttt{sqlite3} module
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{GUI $\rightarrow$ Database Flow}
  \begin{itemize}[<+->]
    \item User enters values in GUI
    \item App validates input
    \item App runs SQL query (INSERT/SELECT/UPDATE/DELETE)
    \item App shows updated view (Listbox/Table)
  \end{itemize}
\end{frame}

\section{Summary}
\label{sec:summary}

\begin{frame}[fragile]{Key Takeaways}
  \begin{itemize}[<+->]
    \item Databases provide structured storage, querying, and integrity
    \item Relational DBs use tables and SQL; NoSQL offers flexible models
    \item SQLite is a practical starting point using Python's \texttt{sqlite3}
    \item GUI apps typically follow input $\rightarrow$ validate $\rightarrow$ query $\rightarrow$ refresh view
  \end{itemize}
\end{frame}
