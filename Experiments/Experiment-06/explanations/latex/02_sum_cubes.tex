\section*{Objective}
Compute the sum of cubes of positive integers smaller than \texttt{n}.
\section*{Input}
An integer \texttt{n}.
\section*{Output}
Sum of cubes from 1 to \texttt{n-1}.
\section*{Approach}
Loop from 1 to \texttt{n-1} and accumulate \(i^3\).
\section*{Time Complexity}
\texttt{O(n)}.
